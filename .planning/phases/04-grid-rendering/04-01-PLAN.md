    ```go
    // GridData is the view model for the grid template.
    type GridData struct {
        ShelfName  string
        Rows       int
        Cols       int
        ColNumbers []int  // [1, 2, 3, ..., Cols] for column header iteration
        Grid       []Row
        Error      string // non-empty if shelf could not be loaded (per D-07)
    }

    // Row represents one row in the grid display.
    type Row struct {
        Letter string // "A", "B", "C", ...
        Cells  []Cell
    }

    // Cell represents one container position.
    type Cell struct {
        Coord    string // from labelFor(), e.g. "3B"
        Col      int    // 1-based
        Row      int    // 1-based
        Count    int    // number of items
        IsEmpty  bool   // true when Count == 0
        CSSClass string // "cell-light" or "cell-dark"
    }
    ```

    2. In store.go, add method `func (s *Store) GetGridData() (*GridData, error)`:
       - Query the first shelf: `SELECT id, name, rows, cols FROM shelf LIMIT 1`
       - Query containers with item counts:
         ```sql
         SELECT c.col, c.row, COUNT(i.id) AS item_count
         FROM container c
         LEFT JOIN item i ON c.id = i.container_id
         WHERE c.shelf_id = ?
         GROUP BY c.id, c.col, c.row
         ORDER BY c.row, c.col
         ```
       - Build a `map[pos]int` (where `type pos struct{ row, col int }`) from results
       - Build `ColNumbers` slice: `[]int{1, 2, ..., shelf.Cols}`
       - Build nested `[]Row` structure:
         - Outer loop `r` from 1 to shelf.Rows, each Row.Letter = `string(rune('A' + r - 1))`
         - Inner loop `c` from 1 to shelf.Cols, each Cell:
           - Coord = `labelFor(c, r)`
           - Count from map lookup (0 if missing)
           - IsEmpty = count == 0
           - CSSClass: `"cell-light"` if `(c+r)%2 == 0`, else `"cell-dark"`
       - Return `&GridData{ShelfName: shelf.Name, Rows: shelf.Rows, Cols: shelf.Cols, ColNumbers: colNums, Grid: grid, Error: ""}`, nil
       - The `pos` type should be declared as unexported inside store.go, right above GetGridData
