    ```go
    func handleResizeShelf(store *Store) http.HandlerFunc {
        return func(w http.ResponseWriter, r *http.Request) {
            var req ResizeRequest
            if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
                writeError(w, http.StatusBadRequest, "invalid JSON")
                return
            }
            if msg := validateResizeRequest(&req); msg != "" {
                writeError(w, http.StatusBadRequest, msg)
                return
            }
            result, err := store.ResizeShelf(r.Context(), req.Rows, req.Cols)
            if err != nil {
                slog.Error("resize shelf", "err", err)
                writeError(w, http.StatusInternalServerError, "internal error")
                return
            }
            if result.Blocked {
                writeJSON(w, http.StatusConflict, result)
                return
            }
            // If name provided, update shelf name
            if req.Name != nil && *req.Name != "" {
                // Update shelf name via a simple query
                store.db.ExecContext(r.Context(),
                    "UPDATE shelf SET name = ?, updated_at = datetime('now') WHERE id = (SELECT id FROM shelf LIMIT 1)",
                    *req.Name)
            }
            writeJSON(w, http.StatusOK, result)
        }
    }
    ```
